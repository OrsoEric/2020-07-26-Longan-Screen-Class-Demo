417 bool Display::init( void )
418 {
419  //----------------------------------------------------------------
420  // BODY
421  //----------------------------------------------------------------
422 
423  rcu_periph_clock_enable( RCU_GPIOA );
424  rcu_periph_clock_enable( RCU_GPIOB );
425  rcu_periph_clock_enable( RCU_AF );
426  rcu_periph_clock_enable( RCU_DMA0 );
427  rcu_periph_clock_enable( RCU_SPI0 );
428 
429  gpio_init(GPIOA, GPIO_MODE_AF_PP, GPIO_OSPEED_50MHZ, GPIO_PIN_5 | GPIO_PIN_7);
430  gpio_init(GPIOB, GPIO_MODE_OUT_PP, GPIO_OSPEED_50MHZ, GPIO_PIN_0 | GPIO_PIN_1 | GPIO_PIN_2);
431 
432  this -> rs_mode_data();
433  this -> rst_active();
434  this -> cs_inactive();
435 
436  Longan_nano::Chrono::delay( Longan_nano::Chrono::Unit::milliseconds, 1 );
437  this -> rst_inactive();
438 
439  Longan_nano::Chrono::delay( Longan_nano::Chrono::Unit::milliseconds, 5 );
440 
441  // Deinit SPI and DMA.
442  spi_i2s_deinit(Config::SPI_CH);
443  dma_deinit(DMA0, DMA_CH1);
444  dma_deinit(DMA0, DMA_CH2);
445 
446  // Configure DMA, do not enable.
447  DMA_CHCTL(DMA0, DMA_CH1) = (uint32_t)(DMA_PRIORITY_ULTRA_HIGH | DMA_CHXCTL_MNAGA); // Receive.
448  DMA_CHCTL(DMA0, DMA_CH2) = (uint32_t)(DMA_PRIORITY_ULTRA_HIGH | DMA_CHXCTL_DIR); // Transmit.
449  DMA_CHPADDR(DMA0, DMA_CH1) = (uint32_t)&SPI_DATA(Config::SPI_CH);
450  DMA_CHPADDR(DMA0, DMA_CH2) = (uint32_t)&SPI_DATA(Config::SPI_CH);
451 
452  // Configure and enable SPI.
453  SPI_CTL0(Config::SPI_CH) = (uint32_t)(SPI_MASTER | SPI_TRANSMODE_FULLDUPLEX | SPI_FRAMESIZE_8BIT | SPI_NSS_SOFT | SPI_ENDIAN_MSB | SPI_CK_PL_LOW_PH_1EDGE | SPI_PSC_8);
454  SPI_CTL1(Config::SPI_CH) = (uint32_t)(SPI_CTL1_DMATEN);
455  spi_enable(Config::SPI_CH);
456 
457  // Enable lcd controller.
458  this -> cs_active();
459 
460  // Initialize the display.
461  for (const uint8_t* p = this -> g_st7735s_init_sequence; *p != Command::TERMINATOR; p++)
462  {
463  this -> spi_wait_idle();
464  this -> spi_set_8bit();
465  this -> rs_mode_cmd();
466  spi_i2s_data_transmit(Config::SPI_CH, *p++);
467 
468  if (*p == Command::TERMINATOR)
469  {
470  continue;
471  }
472 
473  this -> spi_wait_idle();
474  this -> rs_mode_data();
475 
476  while(*p != Command::TERMINATOR)
477  {
478  this -> spi_wait_tbe();
479  spi_i2s_data_transmit(Config::SPI_CH, *p++);
480  }
481  }
482 
483  //----------------------------------------------------------------
484  // RETURN
485  //----------------------------------------------------------------
486 
487  return false;
488 } //End Public init: init | void |

606 
607 bool Display::update_sprite( void )
608 {
609  //----------------------------------------------------------------
610  // BODY
611  //----------------------------------------------------------------
612 
613  //Switch: FSM status
614  switch (this -> g_sprite_status)
615  {
616  //IDLE
617  case 0:
618  {
619  //Do Nothing
620 
621  break;
622  }
623  //Row address
624  case 1:
625  {
626  if ((SPI_STAT( Config::SPI_CH ) &SPI_STAT_TRANS) == 0)
627  {
628  this -> spi_set_8bit();
629  this -> rs_mode_cmd();
630  spi_i2s_data_transmit(Config::SPI_CH, Command::SEND_ROW_ADDRESS);
631  //Next state
632  this -> g_sprite_status++;
633  }
634  break;
635  }
636  //Row Address start
637  case 2:
638  {
639  //If: user wants to use the DMA
640  if (Config::USE_DMA == true)
641  {
642  if ((SPI_STAT( Config::SPI_CH ) &SPI_STAT_TRANS) == 0)
643  {
644  //Configure HW
645  this -> rs_mode_data();
646  this -> spi_set_16bit();
647  //Load addresses on the address buffer
648  this -> g_address_buffer[ 0 ] = this -> g_sprite.origin_w +Config::ROW_ADDRESS_OFFSET;
649  this -> g_address_buffer[ 1 ] = this -> g_sprite.origin_w +Config::ROW_ADDRESS_OFFSET +this -> g_sprite.size_w -1;
650  //Program the DMA
651  dma_channel_disable( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH );
652  dma_memory_width_config( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH, DMA_MEMORY_WIDTH_16BIT );
653  dma_periph_width_config( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH, DMA_PERIPHERAL_WIDTH_16BIT );
654  dma_memory_address_config( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH, (uint32_t)(this -> g_address_buffer) );
655  dma_memory_increase_enable( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH );
656  dma_transfer_number_config( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH, 2 );
657  //Begin the DMA transfer
658  dma_channel_enable( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH );
659  //Next state
660  this -> g_sprite_status += 2;
661  }
662  }
663  //End If: user doesn't want to use the DMA
664  else
665  {
666  if ((SPI_STAT( Config::SPI_CH ) &SPI_STAT_TRANS) == 0)
667  {
668  this -> spi_set_16bit();
669  this -> rs_mode_data();
670  spi_i2s_data_transmit(Config::SPI_CH, this -> g_sprite.origin_w +Config::ROW_ADDRESS_OFFSET);
671  //Next state
672  this -> g_sprite_status++;
673  }
674  }
675  break;
676  }
677  //Row Address stop
678  case 3:
679  {
680  if ((SPI_STAT( Config::SPI_CH ) &SPI_STAT_TBE) != 0)
681  {
682  spi_i2s_data_transmit(Config::SPI_CH, this -> g_sprite.origin_w +Config::ROW_ADDRESS_OFFSET +this -> g_sprite.size_w -1);
683  //Next state
684  this -> g_sprite_status++;
685  }
686  break;
687  }
688  //Col Address
689  case 4:
690  {
691  if ((SPI_STAT( Config::SPI_CH ) &SPI_STAT_TRANS) == 0)
692  {
693  this -> spi_set_8bit();
694  this -> rs_mode_cmd();
695  spi_i2s_data_transmit(Config::SPI_CH, Command::SEND_COL_ADDRESS);
696  //Next state
697  this -> g_sprite_status++;
698  }
699  break;
700  }
701  //Col Address Start
702  case 5:
703  {
704  //If: user wants to use the DMA
705  if (Config::USE_DMA == true)
706  {
707  if ((SPI_STAT( Config::SPI_CH ) &SPI_STAT_TRANS) == 0)
708  {
709  //Configure HW
710  this -> rs_mode_data();
711  this -> spi_set_16bit();
712  //Load addresses on the address buffer
713  this -> g_address_buffer[ 0 ] = this -> g_sprite.origin_h +Config::COL_ADDRESS_OFFSET;
714  this -> g_address_buffer[ 1 ] = this -> g_sprite.origin_h +Config::COL_ADDRESS_OFFSET +this -> g_sprite.size_h -1;
715  //Program the DMA
716  dma_channel_disable( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH );
717  dma_memory_width_config( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH, DMA_MEMORY_WIDTH_16BIT );
718  dma_periph_width_config( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH, DMA_PERIPHERAL_WIDTH_16BIT );
719  dma_memory_address_config( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH, (uint32_t)(this -> g_address_buffer) );
720  dma_memory_increase_enable( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH );
721  dma_transfer_number_config( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH, 2 );
722  //Begin the DMA transfer
723  dma_channel_enable( Config::DMA_SPI_TX, (dma_channel_enum)Config::DMA_SPI_TX_CH );
724  //Next state
725  this -> g_sprite_status += 2;
726  }
727  }
728  //End If: user doesn't want to use the DMA
729  else
730  {
