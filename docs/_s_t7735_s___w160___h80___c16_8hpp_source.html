226  typedef enum _Command
227  {
228  // SETUP_NORMAL_MODE
229  //Frame rate=850kHz/((RTNA x 2 + 40) x (LINE + FPA + BPA +2))
230  //Frame Rate = 850Khz/( (5*2+40)*(80+56+56+2) )= 87.6[Hz]
231  //RTNA | FPA | BPA
232  SETUP_NORMAL_MODE = 0xB1,
233  // SETUP_IDLE_MODE
234  //Frame rate=850kHz/((RTNA x 2 + 40) x (LINE + FPA + BPA +2))
235  //Frame Rate = 850Khz/( (5*2+40)*(80+56+56+2) )= 87.6[Hz]
236  //RTNA | FPA | BPA
237  SETUP_IDLE_MODE = 0xB2,
238 
239  SETUP_PARTIAL_MODE = 0xB3,
240  // DISPLAY_INVERSION_CONTROL
241  //false Dot Inversion | true = Normal Mode Column Inversion
242  //Bit 0 | Normal Mode
243  //Bit 1 | idle Mode
244  //Bit 2 | partial mode/full colors
245  DISPLAY_INVERSION_CONTROL = 0xB4,
246 
247  POWER_GVDD = 0xC0,
248  POWER_VGH_VGL = 0xC1,
249  POWER_VCOM1 = 0xC5,
250 
251  POWER_MODE_NORMAL = 0xC2,
252  POWER_MODE_IDLE = 0xC3,
253  POWER_MODE_PARTIAL = 0xC4,
254 
255  ADJUST_GAMMA_PLUS = 0xE0,
256  ADJUST_GAMMA_MINUS = 0xE1,
257 
258  // COLOR_FORMAT
259  //0x03 | 12b |4R4G|4B...4R|4G4B| 3 bytes transfer 2 color
260  //0x05 | 16b |5R3G|3G5B| 2 bytes transfer 1 color
261  //0x06 | 18b |6R..|6G..|6b..| 3 bytes transfer 1 color
262  COLOR_FORMAT = 0x3A,

632 bool Display::update_sprite( void )
633 {
634  //----------------------------------------------------------------
635  // BODY
636  //----------------------------------------------------------------
637 
638  //Switch: FSM status
639  switch (this -> g_sprite_status)
640  {
641  //IDLE
642  case 0:
643  {
644  //Do Nothing
645 
646  break;
647  }
648  //Row address
649  case 1:
650  {
651  if (this -> is_spi_idle() == true)
652  {
653  //Commands are 8b
654  this -> spi_set_8bit();
655  this -> rs_mode_cmd();
656  spi_i2s_data_transmit(Config::SPI_CH, Command::SEND_ROW_ADDRESS);
657  //Next state
658  this -> g_sprite_status++;
659  }
660  break;
661  }
662  //Row Address start
663  case 2:
664  {
665  //If: user wants to use the DMA
666  if (Config::USE_DMA == true)
667  {
668  if (this -> is_spi_idle() == true)
669  {
670  //Configure Data are 16b
671  this -> rs_mode_data();
672  this -> spi_set_16bit();
673  //Load addresses on the address buffer
674  this -> g_address_buffer[ 0 ] = this -> g_sprite.origin_w +Config::ROW_ADDRESS_OFFSET;
675  this -> g_address_buffer[ 1 ] = this -> g_sprite.origin_w +Config::ROW_ADDRESS_OFFSET +this -> g_sprite.size_w -1;
676  //Program the DMA to send the address
677  this -> dma_send_map16( this -> g_address_buffer, 2 );
678  //Next state. Skip second SPI transfer
679  this -> g_sprite_status += 2;
680  }
681  }
682  //End If: user doesn't want to use the DMA
683  else
684  {
685  if (this -> is_spi_idle() == true)
686  {
687  this -> spi_set_16bit();
688  this -> rs_mode_data();
689  spi_i2s_data_transmit(Config::SPI_CH, this -> g_sprite.origin_w +Config::ROW_ADDRESS_OFFSET);
690  //Next state
691  this -> g_sprite_status++;
692  }
693  }
694  break;
695  }
696  //Row Address stop
697  case 3:
698  {
699  if (this -> is_spi_done_tx() == true)
700  {
701  spi_i2s_data_transmit(Config::SPI_CH, this -> g_sprite.origin_w +Config::ROW_ADDRESS_OFFSET +this -> g_sprite.size_w -1);
702  //Next state
703  this -> g_sprite_status++;
704  }
705  break;
706  }
707  //Col Address
708  case 4:
709  {
710  if (this -> is_spi_idle() == true)
711  {
712  //Commands are 8b
713  this -> spi_set_8bit();
714  this -> rs_mode_cmd();
715  spi_i2s_data_transmit(Config::SPI_CH, Command::SEND_COL_ADDRESS);
716  //Next state
717  this -> g_sprite_status++;
718  }
719  break;
720  }
721  //Col Address Start
722  case 5:
723  {
724  //If: user wants to use the DMA
725  if (Config::USE_DMA == true)
726  {
727  if (this -> is_spi_idle() == true)
728  {
729  //Configure HW
730  this -> rs_mode_data();
731  this -> spi_set_16bit();
732  //Load addresses on the address buffer
733  this -> g_address_buffer[ 0 ] = this -> g_sprite.origin_h +Config::COL_ADDRESS_OFFSET;
734  this -> g_address_buffer[ 1 ] = this -> g_sprite.origin_h +Config::COL_ADDRESS_OFFSET +this -> g_sprite.size_h -1;
735  //Program the DMA to send the address
736  this -> dma_send_map16( this -> g_address_buffer, 2 );
737  //Next state. Skip second SPI transfer
738  this -> g_sprite_status += 2;
739  }
740  }
741  //End If: user doesn't want to use the DMA
742  else
743  {
744  if (this -> is_spi_idle() == true)
745  {
746  this -> spi_set_16bit();
747  this -> rs_mode_data();
748  spi_i2s_data_transmit(Config::SPI_CH, this -> g_sprite.origin_h +Config::COL_ADDRESS_OFFSET);
749  //Next state
750  this -> g_sprite_status++;
751  }
752  }
753  break;
754  }
755  //Col Address Stop
756  case 6:
757  {
758  if (this -> is_spi_done_tx() == true)
759  {
760  spi_i2s_data_transmit( Config::SPI_CH, this -> g_sprite.origin_h +Config::COL_ADDRESS_OFFSET +this -> g_sprite.size_h -1 );
761  //Next state
762  this -> g_sprite_status++;
763  }
764  break;
765  }
766  //Write Memory
767  case 7:
768  {
769  if (this -> is_spi_idle() == true)
770  {
771  this -> spi_set_8bit();
772  this -> rs_mode_cmd();
773  spi_i2s_data_transmit( Config::SPI_CH, Command::WRITE_MEM );
774  //Next state. DMA uses a single state, SPI use one state per pixel
775  this -> g_sprite_status = ((Config::USE_DMA == true)?(8):(10));
776  }
777  break;
778  }
779  //DMA Send
780  case 8:
781  {
782  if (this -> is_spi_idle() == true)
783  {
